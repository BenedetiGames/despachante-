// ============================================================ // SISTEMA DESPACHANTE — Google Apps Script v2 (com usuários) // ============================================================ var SHEET_CREDITOS = "CREDITOS"; var SHEET_NFS = "NFS"; var SHEET_LOJAS = "LOJAS"; var SHEET_VENDEDORES = "VENDEDORES"; var SHEET_USUARIOS = "USUARIOS"; var HEADERS_CREDITOS = ["id","cliente","chassi","placa","oc","ipva","loja","vendedor","tipo","valor","comissao","honorarios","nossaCom","status","motivo","obs","dtVenda","semana"]; var HEADERS_NFS = ["id","cliente","chassi","loja","dtFat","dtReceb","obs"]; var HEADERS_LOJAS = ["nome"]; var HEADERS_VENDEDORES = ["nome"]; var HEADERS_USUARIOS = ["id","nome","usuario","senha","perfil","lojas"]; // perfil: "adm" ou "usuario" // lojas: nomes separados por vírgula ex: "JEEP,FIAT,BYD" // ------------------------------------------------------------ // ENTRADA // ------------------------------------------------------------ function doGet(e) { try { var action = e && e.parameter && e.parameter.action ? e.parameter.action : ""; var sheet = e && e.parameter && e.parameter.sheet ? e.parameter.sheet : ""; var result; if (action === "init") result = initSheets(); else if (action === "getAll") result = getAll(sheet); else result = { status: "ok", message: "Sistema Despachante v2 ativo!" }; return resposta(result); } catch(err) { return resposta({ error: err.toString() }); } } function doPost(e) { try { var body = JSON.parse(e.postData.contents); var action = body.action; var result; if (action === "init") result = initSheets(); else if (action === "getAll") result = getAll(body.sheet); else if (action === "insert") result = insert(body.sheet, body.data); else if (action === "delete") result = deleteRow(body.sheet, body.id); else if (action === "replaceAll") result = replaceAll(body.sheet, body.data); else if (action === "login") result = login(body.usuario, body.senha); else if (action === "saveUsuario") result = saveUsuario(body.data); else if (action === "deleteUsuario") result = deleteUsuario(body.id); else result = { error: "Acao desconhecida: " + action }; return resposta(result); } catch(err) { return resposta({ error: err.toString() }); } } function resposta(obj) { return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); } // ------------------------------------------------------------ // LOGIN // ------------------------------------------------------------ function login(usuario, senha) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(SHEET_USUARIOS); if (!sheet) return { error: "Aba USUARIOS não encontrada." }; var values = sheet.getDataRange().getValues(); var headers = values[0]; var rows = values.slice(1); var iUsuario = headers.indexOf("usuario"); var iSenha = headers.indexOf("senha"); var iPerfil = headers.indexOf("perfil"); var iLojas = headers.indexOf("lojas"); var iNome = headers.indexOf("nome"); var iId = headers.indexOf("id"); for (var i = 0; i < rows.length; i++) { if (String(rows[i][iUsuario]).toLowerCase() === String(usuario).toLowerCase() && String(rows[i][iSenha]) === String(senha)) { return { status: "ok", user: { id: rows[i][iId], nome: rows[i][iNome], usuario: rows[i][iUsuario], perfil: rows[i][iPerfil], lojas: rows[i][iLojas] ? String(rows[i][iLojas]).split(",").map(function(l){return l.trim();}) : [] } }; } } return { error: "Usuário ou senha incorretos." }; } // ------------------------------------------------------------ // SALVAR USUÁRIO (insert ou update) // ------------------------------------------------------------ function saveUsuario(data) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(SHEET_USUARIOS); if (!sheet) return { error: "Aba USUARIOS não encontrada." }; var values = sheet.getDataRange().getValues(); var headers = values[0]; var iId = headers.indexOf("id"); // Se tem id, tenta atualizar linha existente if (data.id) { for (var i = 1; i < values.length; i++) { if (String(values[i][iId]) === String(data.id)) { var row = headers.map(function(h) { if (h === "lojas" && Array.isArray(data[h])) return data[h].join(","); return data[h] !== undefined ? data[h] : ""; }); sheet.getRange(i + 1, 1, 1, row.length).setValues([row]); return { status: "ok", action: "updated" }; } } } // Senão insere novo data.id = Date.now(); var row = headers.map(function(h) { if (h === "lojas" && Array.isArray(data[h])) return data[h].join(","); return data[h] !== undefined ? data[h] : ""; }); sheet.insertRowAfter(1); sheet.getRange(2, 1, 1, row.length).setValues([row]); return { status: "ok", action: "inserted", id: data.id }; } // ------------------------------------------------------------ // DELETAR USUÁRIO // ------------------------------------------------------------ function deleteUsuario(id) { return deleteRow(SHEET_USUARIOS, id); } // ------------------------------------------------------------ // INIT // ------------------------------------------------------------ function initSheets() { var ss = SpreadsheetApp.getActiveSpreadsheet(); criarAba(ss, SHEET_CREDITOS, HEADERS_CREDITOS); criarAba(ss, SHEET_NFS, HEADERS_NFS); criarAba(ss, SHEET_LOJAS, HEADERS_LOJAS); criarAba(ss, SHEET_VENDEDORES, HEADERS_VENDEDORES); criarAba(ss, SHEET_USUARIOS, HEADERS_USUARIOS); // Cria adm padrão se não houver nenhum usuário var sheetU = ss.getSheetByName(SHEET_USUARIOS); if (sheetU.getLastRow() <= 1) { sheetU.appendRow([Date.now(), "Administrador", "adm", "1234", "adm", ""]); } return { status: "ok", message: "Abas criadas. Login padrão: usuario=adm / senha=1234" }; } function criarAba(ss, nome, headers) { var sheet = ss.getSheetByName(nome); if (!sheet) { sheet = ss.insertSheet(nome); sheet.getRange(1, 1, 1, headers.length).setValues([headers]); sheet.getRange(1, 1, 1, headers.length).setFontWeight("bold").setBackground("#1D9E75").setFontColor("#ffffff"); sheet.setFrozenRows(1); } return sheet; } // ------------------------------------------------------------ // GET ALL // ------------------------------------------------------------ function getAll(sheetName) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(sheetName); if (!sheet) return { data: [] }; var values = sheet.getDataRange().getValues(); if (values.length <= 1) return { data: [] }; var headers = values[0]; var rows = values.slice(1).map(function(row) { var obj = {}; headers.forEach(function(h, i) { if (row[i] === "TRUE" || row[i] === true) obj[h] = true; else if (row[i] === "FALSE" || row[i] === false) obj[h] = false; else obj[h] = row[i]; }); return obj; }); return { data: rows }; } // ------------------------------------------------------------ // INSERT // ------------------------------------------------------------ function insert(sheetName, data) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(sheetName); if (!sheet) return { error: "Aba nao encontrada: " + sheetName }; var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; var row = headers.map(function(h) { var val = data[h]; if (val === undefined || val === null) return ""; if (typeof val === "boolean") return val ? "TRUE" : "FALSE"; return val; }); sheet.insertRowAfter(1); sheet.getRange(2, 1, 1, row.length).setValues([row]); return { status: "ok", id: data.id }; } // ------------------------------------------------------------ // DELETE // ------------------------------------------------------------ function deleteRow(sheetName, id) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(sheetName); if (!sheet) return { error: "Aba nao encontrada: " + sheetName }; var values = sheet.getDataRange().getValues(); var headers = values[0]; var idCol = headers.indexOf("id"); if (idCol === -1) return { error: "Coluna id nao encontrada." }; for (var i = 1; i < values.length; i++) { if (String(values[i][idCol]) === String(id)) { sheet.deleteRow(i + 1); return { status: "ok", deleted: id }; } } return { error: "Registro nao encontrado: " + id }; } // ------------------------------------------------------------ // REPLACE ALL // ------------------------------------------------------------ function replaceAll(sheetName, data) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(sheetName); if (!sheet) return { error: "Aba nao encontrada: " + sheetName }; var lastRow = sheet.getLastRow(); if (lastRow > 1) sheet.deleteRows(2, lastRow - 1); if (data && data.length > 0) { var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; var rows = data.map(function(item) { return headers.map(function(h) { return item[h] || ""; }); }); sheet.getRange(2, 1, rows.length, rows[0].length).setValues(rows); } return { status: "ok" }; }
